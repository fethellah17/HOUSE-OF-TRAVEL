/* Inbox - New 4-Tab Specialized System */
import { Plane as PlaneIcon, Hotel as HotelIcon, Map as MapIcon, FileText as FileTextIcon } from "lucide-react";
import type { ServiceRequest, BilletterieRequest, VisaRequest, HotelRequest, SejourRequest } from "@/contexts/DataContext";

const InboxView = ({
  requests,
  markRequestAsRead,
  deleteRequest,
}: {
  requests: ServiceRequest[];
  markRequestAsRead: (id: string) => void;
  deleteRequest: (id: string) => void;
}) => {
  const [activeServiceTab, setActiveServiceTab] = useState<"billetterie" | "visa" | "hotel" | "sejour">("billetterie");
  const [selectedRequest, setSelectedRequest] = useState<ServiceRequest | null>(null);

  // Filter requests by service type
  const billetterieRequests = requests.filter(r => r.serviceType === "billetterie") as BilletterieRequest[];
  const visaRequests = requests.filter(r => r.serviceType === "visa") as VisaRequest[];
  const hotelRequests = requests.filter(r => r.serviceType === "hotel") as HotelRequest[];
  const sejourRequests = requests.filter(r => r.serviceType === "sejour") as SejourRequest[];

  const unreadCount = requests.filter(r => !r.isRead).length;

  const openRequest = (req: ServiceRequest) => {
    setSelectedRequest(req);
    if (!req.isRead) {
      markRequestAsRead(req.id);
    }
  };

  const serviceTabs = [
    { id: "billetterie" as const, label: "Billetterie", icon: PlaneIcon, count: billetterieRequests.length },
    { id: "visa" as const, label: "Visa", icon: FileTextIcon, count: visaRequests.length },
    { id: "hotel" as const, label: "Hôtel", icon: HotelIcon, count: hotelRequests.length },
    { id: "sejour" as const, label: "Séjour à la Carte", icon: MapIcon, count: sejourRequests.length },
  ];

  return (
    <div>
      <h2 className="text-xl sm:text-2xl font-semibold mb-4 sm:mb-6">Boîte de Réception</h2>

      {/* Stats Card */}
      <motion.div
        initial={{ opacity: 0, y: -10 }}
        animate={{ opacity: 1, y: 0 }}
        className="mb-6"
      >
        <div className="bg-white rounded-lg shadow-sm border border-slate-200 p-3">
          <div className="flex items-center justify-between">
            <div>
              <p className="text-xs font-medium text-slate-600 mb-0.5">Demandes Non Lues</p>
              <p className="text-xl font-bold tabular-nums text-slate-900">{unreadCount}</p>
            </div>
            <div className="bg-slate-100 p-2.5 rounded-lg">
              <Inbox size={20} className="text-slate-600" />
            </div>
          </div>
        </div>
      </motion.div>

      {/* Service Tabs */}
      <div className="flex gap-2 mb-6 overflow-x-auto pb-2">
        {serviceTabs.map((tab) => {
          const Icon = tab.icon;
          return (
            <button
              key={tab.id}
              onClick={() => setActiveServiceTab(tab.id)}
              className={`flex items-center gap-2 px-4 py-3 rounded-xl font-semibold transition-all whitespace-nowrap ${
                activeServiceTab === tab.id
                  ? "bg-primary text-white shadow-md"
                  : "bg-white text-slate-700 border border-slate-200 hover:border-primary"
              }`}
            >
              <Icon size={18} />
              <span>{tab.label}</span>
              <span className={`px-2 py-0.5 rounded-full text-xs font-bold ${
                activeServiceTab === tab.id
                  ? "bg-white/20 text-white"
                  : "bg-slate-100 text-slate-600"
              }`}>
                {tab.count}
              </span>
            </button>
          );
        })}
      </div>

      {/* Request Detail Modal */}
      <AnimatePresence>
        {selectedRequest && (
          <motion.div
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            exit={{ opacity: 0 }}
            className="fixed inset-0 z-50 bg-black/20 flex items-center justify-center p-4"
            onClick={() => setSelectedRequest(null)}
          >
            <motion.div
              initial={{ scale: 0.95 }}
              animate={{ scale: 1 }}
              exit={{ scale: 0.95 }}
              className="bg-white rounded-2xl shadow-2xl p-6 max-w-2xl w-full max-h-[90vh] overflow-y-auto"
              onClick={(e) => e.stopPropagation()}
            >
              <div className="flex items-start justify-between mb-6">
                <div>
                  <span className="inline-block px-3 py-1 bg-primary/10 text-primary text-xs font-semibold rounded-full mb-2">
                    {selectedRequest.serviceType.toUpperCase()}
                  </span>
                  <h3 className="text-lg font-bold text-primary">
                    {selectedRequest.personalInfo.prenom} {selectedRequest.personalInfo.nom}
                  </h3>
                </div>
                <button
                  onClick={() => setSelectedRequest(null)}
                  className="text-slate-400 hover:text-slate-600 p-2"
                >
                  <X size={20} />
                </button>
              </div>

              {/* Render details based on service type */}
              {selectedRequest.serviceType === "billetterie" && (
                <BilletterieDetails request={selectedRequest as BilletterieRequest} />
              )}
              {selectedRequest.serviceType === "visa" && (
                <VisaDetails request={selectedRequest as VisaRequest} />
              )}
              {selectedRequest.serviceType === "hotel" && (
                <HotelDetails request={selectedRequest as HotelRequest} />
              )}
              {selectedRequest.serviceType === "sejour" && (
                <SejourDetails request={selectedRequest as SejourRequest} />
              )}

              <div className="mt-6 flex gap-3">
                <button
                  onClick={() => {
                    deleteRequest(selectedRequest.id);
                    setSelectedRequest(null);
                    toast.success("Demande supprimée");
                  }}
                  className="flex-1 px-4 py-2 bg-red-600 text-white rounded-lg hover:bg-red-700 transition-colors"
                >
                  Supprimer
                </button>
                <button
                  onClick={() => setSelectedRequest(null)}
                  className="flex-1 px-4 py-2 bg-slate-100 text-slate-700 rounded-lg hover:bg-slate-200 transition-colors"
                >
                  Fermer
                </button>
              </div>
            </motion.div>
          </motion.div>
        )}
      </AnimatePresence>

      {/* Dynamic Tables/Cards */}
      {activeServiceTab === "billetterie" && (
        <BilletterieTable requests={billetterieRequests} onOpen={openRequest} />
      )}
      {activeServiceTab === "visa" && (
        <VisaTable requests={visaRequests} onOpen={openRequest} />
      )}
      {activeServiceTab === "hotel" && (
        <HotelTable requests={hotelRequests} onOpen={openRequest} />
      )}
      {activeServiceTab === "sejour" && (
        <SejourTable requests={sejourRequests} onOpen={openRequest} />
      )}
    </div>
  );
};

          >
            <div className="flex items-start justify-between mb-3">
              <div>
                <p className="font-semibold text-primary">{req.personalInfo.prenom} {req.personalInfo.nom}</p>
                <p className="text-xs text-slate-500">{req.personalInfo.email}</p>
              </div>
              {!req.isRead && <span className="px-2 py-1 bg-primary text-white text-xs rounded-full">Nouveau</span>}
            </div>
            <div className="space-y-2 text-sm">
              <p><strong>Destination:</strong> {req.destination}</p>
              <p><strong>Type:</strong> {req.tripType}</p>
              <p><strong>Départ:</strong> {new Date(req.dateDepart).toLocaleDateString("fr-FR")}</p>
              <p><strong>Passagers:</strong> {req.nombreAdultes} adulte(s), {req.nombreEnfants} enfant(s)</p>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
};
