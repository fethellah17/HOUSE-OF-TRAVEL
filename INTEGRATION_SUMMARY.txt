## Common Patterns

### Protected Component
```typescript
function MyPage() {
  const { user, isLoading } = useRequireAuth();
  
  if (isLoading) return <Loading />;
  
  return <div>Hello {user?.prenom}</div>;
}
```

### Data Fetching with Loading
```typescript
function VoyagesList() {
  const { voyages, isLoading, error } = useVoyages();
  
  if (isLoading) return <Spinner />;
  if (error) return <Error message={error} />;
  
  return voyages.map(v => <VoyageCard key={v.id} voyage={v} />);
}
```

### Form Submission
```typescript
function BookingForm() {
  const { create, isLoading, error } = useCreateBooking();
  
  const handleSubmit = async (data) => {
    const result = await create(data);
    if (result.success) {
      toast.success("Success!");
    } else {
      toast.error(result.error);
    }
  };
  
  return <form onSubmit={handleSubmit}>...</form>;
}
```

## Error Handling

All services return:
```typescript
{
  success: boolean;
  data?: T;
  error?: string;
}
```

Handle errors consistently:
```typescript
if (!result.success) {
  console.error("Error:", result.error);
  toast.error(result.error || "Something went wrong");
  return;
}

// Use result.data
```
